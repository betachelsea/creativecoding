<script type="text/ruby">
CANVAS_HEIGHT = 750
CANVAS_WIDTH = 750

INCREASE = 1
DECREASE = 2
G = 0.01
FRAMERATE = 60
# 60
H = 300
OMEGA = (Math::PI * 2) / 3.0 # 1.5秒で1バウンス # 混乱するので整理
# OMEGA = Math::PI * 2
MIN_SCALE = 0.4
MAX_SCALE = 1.6

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  frameRate(FRAMERATE)
  $phase = 0
  $frame = 0
  $percent = 1.0
  $mode = DECREASE
  $h = 30
end

def lerp(a, b, t) # 線形補間
  a + (b - a) * t
end

def draw
  clear
  $frame += 1
  strokeWeight(0);
  fill('gray');
  dt = deltaTime / 1000.0 # milliseconds
  $phase += dt * OMEGA
  $phase %= (Math::PI * 2) # 2πを超えないようにする

   u = Math.sin($phase).abs
   h = H * (u ** 0.7)
   hn = h / H

   $percent = lerp(MIN_SCALE, MAX_SCALE, hn ** 0.8)

  ellipse(200, 400, 100 * $percent, 60 * $percent); # 楕円
  text("h: #{h}", 50, 30);
  text("frameCount: #{$frame}", 50, 50);
  text("percent: #{$percent}", 50, 100);
  text("phase: #{$phase}", 50, 150);
end

def mouseMoved
end

def mouseClicked
  $circles << Circle.new(x: mouseX, y: mouseY, radius: 30, color: $next_color)
  $next_color = Color.new
  $next_color.set_random_color
end

P5::init()
</script>
