<script type="text/ruby">
CANVAS_HEIGHT = 750
CANVAS_WIDTH = 750

class Color
  attr :r, :g, :b

  def set_random_color
    @r = rand(0..255)
    @g = rand(0..255)
    @b = rand(0..255)
  end

  def setup
    fill(r, g, b)
  end
end

class Circle
  attr :x, :y
  attr :radius
  attr :color

  def initialize(x:, y:, radius:, color:)
    @x = x
    @y = y
    @radius = radius
    @color = color
  end

  def draw
    color.setup
    circle(x, y, radius)
  end
end

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  $circles = []
  $next_color = Color.new
  $next_color.set_random_color
  $t = 0
end

def draw
  clear
  $t = $t + 1
  strokeWeight(0);
  ellipse(100,200, 50, 30); # 楕円
end

def mouseMoved
end

def mouseClicked
  $circles << Circle.new(x: mouseX, y: mouseY, radius: 30, color: $next_color)
  $next_color = Color.new
  $next_color.set_random_color
end

P5::init()
</script>
