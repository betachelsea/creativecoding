<script type="text/ruby">
CANVAS_HEIGHT = 750
CANVAS_WIDTH = 750

INCREASE = 1
DECREASE = 2

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  $frame = 0
  $percent = 1.0
  $mode = DECREASE
end

def draw
  clear
  $frame += 1
  strokeWeight(0);
  fill('gray');
  if $mode == DECREASE
    $percent -= 0.01
    $mode = INCREASE if $percent < 0.0
  else
    $percent += 0.01
    $mode = DECREASE if 1.0 < $percent
  end

  ellipse(200, 400, 100 * $percent, 60 * $percent); # 楕円
  text("frameCount: #{$frame}", 50, 50);
end

def mouseMoved
end

def mouseClicked
  $circles << Circle.new(x: mouseX, y: mouseY, radius: 30, color: $next_color)
  $next_color = Color.new
  $next_color.set_random_color
end

P5::init()
</script>
