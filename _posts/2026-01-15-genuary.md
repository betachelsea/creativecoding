<script type="text/ruby">
CANVAS_HEIGHT = 750
CANVAS_WIDTH = 750

INCREASE = 1
DECREASE = 2
G = 0.01
FRAMERATE = 60

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  frameRate(FRAMERATE)
  $frame = 0
  $percent = 1.0
  $mode = DECREASE
end

def draw
  clear
  $frame += 1
  strokeWeight(0);
  fill('gray');
  now_t = 1.0/FRAMERATE * $frame
  if $mode == DECREASE
    # $percent -= 0.01
    $percent -= (G * now_t * now_t) / 2.0
    if $percent < 0.05
        $mode = INCREASE
        $frame = 0
    end
  else
    $percent += (G * now_t * now_t) / 2.0
    # $percent += 0.01
    if 1.0 < $percent
        $mode = DECREASE
        $frame = 0
    end
  end

   text((G * now_t * now_t) /2.0, 50, 150)
   text(G, 50, 200)
   text(now_t, 50, 250)
  # h = g * t * t

  ellipse(200, 400, 100 * $percent, 60 * $percent); # 楕円
  text("frameCount: #{$frame}", 50, 50);
  text("percent: #{$percent}", 50, 100);
end

def mouseMoved
end

def mouseClicked
  $circles << Circle.new(x: mouseX, y: mouseY, radius: 30, color: $next_color)
  $next_color = Color.new
  $next_color.set_random_color
end

P5::init()
</script>
