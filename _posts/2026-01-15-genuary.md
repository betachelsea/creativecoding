<script type="text/ruby">
CANVAS_HEIGHT = 300
CANVAS_WIDTH = 750

class Color
  attr :r, :g, :b

  def set_random_color
    @r = rand(0..255)
    @g = rand(0..255)
    @b = rand(0..255)
  end

  def setup
    fill(r, g, b)
  end
end

class Circle
  attr :x, :y
  attr :radius
  attr :color

  def initialize(x:, y:, radius:, color:)
    @x = x
    @y = y
    @radius = radius
    @color = color
  end

  def draw
    color.setup
    circle(x, y, radius)
  end
end

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  $circles = []
  $next_color = Color.new
  $next_color.set_random_color
end

def draw
  clear
  $circles.each {|c| c.draw }
  $next_color.setup
  circle(mouseX, mouseY, 30)
  circle(10,10,5)
  # ラスト1個だけ上書き描画する 押した後の感触の都合で
  if ($circles[-1])
    $circles[-1].draw
  end
  strokeWeight(0);
   ellipse(200, 150, 50, 30); # 楕円
end

def mouseMoved
end

def mouseClicked
  $circles << Circle.new(x: mouseX, y: mouseY, radius: 30, color: $next_color)
  $next_color = Color.new
  $next_color.set_random_color
end

P5::init()
</script>
