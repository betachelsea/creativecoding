<script type="text/ruby">
# ref: https://processing.org/reference/mouseMoved_.html

def setup
  createCanvas(720, 400)
  $value = 0
end

def draw
  clear
  fill($value)
  rect(mouseX - 25, mouseY - 25, 50, 50)
end

def mouseMoved
  $value = $value + 5
  if ($value > 255)
    $value = 0
  end
end

P5::init()
</script>
