<script type="text/ruby">
CANVAS_HEIGHT = 750
CANVAS_WIDTH = 750
FRAMERATE = 60

class Arrow
    def initialize
        @cx = rand(-50..800)
        @cy = rand(-50..800)
        @th = rand(0..(2 * Math::PI))
        @length = 20
        @per_second = rand(10.0..100.0)
    end

    def move(dt)
        distance = @per_second * dt
        @cx = @cx + distance * Math.cos(@th)
        @cy = @cy + distance * Math.sin(@th)
    end

    def draw
        px = @cx + @length * Math.cos(@th)
        py = @cy + @length * Math.sin(@th)
        line(@cx, @cy, px, py) # 真ん中

        rx = @cx + (@length/2.0) * Math.cos(@th + Math::PI * 3/4)
        ry = @cy + (@length/2.0) * Math.sin(@th + Math::PI * 3/4)
        line(@cx, @cy, rx, ry) # 右

        lx = @cx + (@length/2.0) * Math.cos(@th - Math::PI * 3/4)
        ly = @cy + (@length/2.0) * Math.sin(@th - Math::PI * 3/4)
        line(@cx, @cy, lx, ly) # 左

        line(rx, ry, px, py)
        line(lx, ly, px, py)
    end
end

def setup
  createCanvas(CANVAS_WIDTH, CANVAS_HEIGHT)
  frameRate(FRAMERATE)
  $arrows = 100.times.map { Arrow.new }
end

def draw
  clear
  dt = deltaTime / 1000.0 # milliseconds
  $arrows.each do |arrow|
    arrow.move(dt)
    arrow.draw
  end
  text("dt:#{dt}", 20, 20)
end

P5::init()
</script>
